
{# MACROS: #}

{% macro tableFormat(arr, delim) %}
  {% for item in arr[0] %}{% if loop.first %}{{delim}} l {% else %}{{delim}} c {% endif %}{% if loop.last %}{{delim}} r {{delim}}{% endif %}{% endfor %}
{% endmacro %}

{% macro placeTable(arr, caption, label) %}
  {% import _self as body %}
  \begin{table}[!ht]
  \captionsetup{font=large}
  \centering
  \caption{ {{ caption }} }
  \label{{ '{tab:'~ label ~'}' }}
  \begin{tabular}{ {{body.tableFormat(arr, '|')}} }
  \hline

  {% for row in arr %}
  {% for column in row %}
  {{column}} {% if not loop.last %}&{% endif %}
  {% endfor %}
  \\ \hline
  {% endfor %}

  \end{tabular}
  \end{table}

{% endmacro %}

{% macro image(file, caption, size, ref) %}
  \begin{figure}[!ht]
  \centering
  \caption{{ '{' ~ caption ~ '\label{fig:' ~ref~ '}}' }}
  \includegraphics[width={{size}}\textwidth]{{ '{'~ file ~'}'}}
  \end{figure}
{% endmacro %}
